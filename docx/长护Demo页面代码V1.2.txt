            const progressPercent = (step - 1) * 25;

            // 更新进度条宽度
            progressBar.style.setProperty('--progress-width', `${progressPercent}%`);
            progressBar.style.width = `${progressPercent}%`;
            progressBar.style.backgroundColor = '#1e6bb8';
            
            // 更新步骤状态
            progressSteps.forEach((stepItem, index) => {
                if (index < step) {
                    stepItem.classList.add('active');
                } else {
                    stepItem.classList.remove('active');
                }
            });

            // 兼容样式
            progressBar.style.setProperty('width', '100%');
            progressBar.style.setProperty('backgroundColor', 'transparent');
            progressBar.style.setProperty('--progress-width', '');
            progressBar.style.after = `width: ${progressPercent}%`;
        }

        // 文件上传处理
        function handleFileUpload(e, type) {
            const files = Array.from(e.target.files);
            if (files.length === 0) return;

            let fileList = type === 'selfAssessment' ? selfAssessmentFiles : 
                          type === 'medical' ? medicalFiles : videoFiles;
            let previewEl = document.getElementById(`${type}FilePreview`);
            let emptyTip = document.getElementById(`${type}EmptyTip`);

            // 隐藏空提示
            emptyTip.style.display = 'none';

            // 添加新文件
            files.forEach(file => {
                const fileInfo = {
                    name: file.name,
                    size: formatFileSize(file.size),
                    time: new Date().toLocaleString(),
                    file: file
                };
                fileList.push(fileInfo);
            });

            // 更新预览
            renderFilePreview(type);

            // 清空input值，允许重新上传相同文件
            e.target.value = '';
        }

        // 渲染文件预览
        function renderFilePreview(type) {
            let fileList = type === 'selfAssessment' ? selfAssessmentFiles : 
                          type === 'medical' ? medicalFiles : videoFiles;
            let previewEl = document.getElementById(`${type}FilePreview`);
            
            // 清空预览区
            previewEl.innerHTML = '';

            // 添加文件项
            fileList.forEach((file, index) => {
                const fileItem = document.createElement('div');
                fileItem.className = 'file-item';
                fileItem.innerHTML = `
                    <div class="file-info">
                        <div class="file-name">${file.name}</div>
                        <div class="file-meta">大小：${file.size} | 上传时间：${file.time}</div>
                    </div>
                    <div class="file-actions">
                        <button class="delete-btn" data-type="${type}" data-index="${index}">删除</button>
                        <button class="reupload-btn" data-type="${type}" data-index="${index}">重新上传</button>
                    </div>
                `;
                previewEl.appendChild(fileItem);
            });

            // 添加删除/重新上传事件
            const deleteBtns = previewEl.querySelectorAll('.delete-btn');
            deleteBtns.forEach(btn => {
                btn.addEventListener('click', function() {
                    const type = this.dataset.type;
                    const index = parseInt(this.dataset.index);
                    deleteFile(type, index);
                });
            });

            const reuploadBtns = previewEl.querySelectorAll('.reupload-btn');
            reuploadBtns.forEach(btn => {
                btn.addEventListener('click', function() {
                    const type = this.dataset.type;
                    const index = parseInt(this.dataset.index);
                    reuploadFile(type, index);
                });
            });

            // 如果没有文件，显示空提示
            if (fileList.length === 0) {
                previewEl.innerHTML = `<div style="text-align: center; color: #666;" id="${type}EmptyTip">暂无上传文件</div>`;
                
                // 如果删除自评表，清空自评结论
                if (type === 'selfAssessment') {
                    const selfAssessmentContents = document.querySelectorAll('.self-assessment-content');
                    selfAssessmentContents.forEach(el => {
                        el.textContent = '暂无自评信息（请先上传客户自评表）';
                    });
                }
            }
        }

        // 删除文件
        function deleteFile(type, index) {
            if (type === 'selfAssessment') {
                selfAssessmentFiles.splice(index, 1);
            } else if (type === 'medical') {
                medicalFiles.splice(index, 1);
            } else {
                videoFiles.splice(index, 1);
            }
            renderFilePreview(type);
        }

        // 重新上传文件
        function reuploadFile(type, index) {
            document.getElementById(`${type}FileInput`).click();
            // 删除原有文件
            deleteFile(type, index);
        }

        // 格式化文件大小
        function formatFileSize(bytes) {
            if (bytes < 1024) return bytes + ' B';
            else if (bytes < 1048576) return (bytes / 1024).toFixed(2) + ' KB';
            else return (bytes / 1048576).toFixed(2) + ' MB';
        }

        // 生成AI建议
        function generateAISuggestion(selectEl) {
            const category = selectEl.dataset.category;
            const item = selectEl.dataset.item;
            const grade = selectEl.value;
            
            // 找到对应的AI建议区域
            const aiContent = selectEl.nextElementSibling.querySelector('.ai-content');
            
            // 模拟AI分析结果
            let medicalSuggestion = '暂无医疗材料，无法生成建议';
            let videoSuggestion = '暂无音视频材料，无法生成建议';
            let selfAssessmentSuggestion = '';

            if (medicalFiles.length > 0) {
                medicalSuggestion = `基于${medicalFiles.length}份医疗材料分析，${item}能力评估为${grade}级，符合${category}评估标准。`;
            }

            if (videoFiles.length > 0) {
                videoSuggestion = `基于${videoFiles.length}份音视频材料观察，被保险人${item}时${grade > 0 ? '需协助' : '可自主完成'}，建议评估等级为${grade}级。`;
            }

            // 新增：加入自评表分析
            if (selfAssessmentFiles.length > 0) {
                selfAssessmentSuggestion = `<div><strong>自评表分析：</strong>客户自评${item}为${selfAssessmentMapping[item] && selfAssessmentMapping[item][grade] ? selfAssessmentMapping[item][grade] : '未填写'}，与评估等级${grade}级${Math.abs(grade - 2) <= 1 ? '基本一致' : '存在差异'}。</div>`;
            }

            // 更新AI建议
            aiContent.innerHTML = `
                <div><strong>医疗材料分析：</strong>${medicalSuggestion}</div>
                <div><strong>音视频材料分析：</strong>${videoSuggestion}</div>
                ${selfAssessmentSuggestion}
            `;
        }

        // 更新评估汇总
        function updateSummary() {
            const summaryContent = document.getElementById('summaryContent');
            let summaryText = `### 长期护理保险失能等级评估结论\n\n`;
            
            // 客户基本信息
            summaryText += `**被保险人信息：**\n`;
            summaryText += `姓名：${document.getElementById('insureeName').value || '未填写'}\n`;
            summaryText += `身份证号：${document.getElementById('idCard').value || '未填写'}\n`;
            summaryText += `评估日期：${document.getElementById('assessmentDate').value || '未填写'}\n`;
            summaryText += `评估人员：${document.getElementById('assessor').value || '未填写'}\n\n`;

            // 新增：自评表信息
            if (selfAssessmentFiles.length > 0) {
                summaryText += `**客户自评信息：**\n`;
                summaryText += `- 已上传自评表：${selfAssessmentFiles[0].name}\n`;
                summaryText += `- 自评综合等级：E级（可申请长期护理失能等级评估）\n\n`;
            }

            // 评估结果汇总
            summaryText += `**各维度评估结果：**\n`;
            const gradeSelects = document.querySelectorAll('.grade-select');
            gradeSelects.forEach(select => {
                const category = select.dataset.category;
                const item = select.dataset.item;
                const grade = select.value;
                const note = select.parentElement.querySelector('.assessment-note').value;
                
                if (grade) {
                    summaryText += `- ${category}-${item}：${grade}级\n`;
                    // 新增：添加自评结论
                    const selfItem = select.dataset.selfItem;
                    if (selfAssessmentFiles.length > 0 && selfAssessmentMapping[selfItem] && selfAssessmentMapping[selfItem][grade]) {
                        summaryText += `  客户自评：${selfAssessmentMapping[selfItem][grade]}\n`;
                    }
                    if (note) summaryText += `  评估意见：${note}\n`;
                }
            });

            // 综合等级计算
            let totalGrade = 0;
            let gradeCount = 0;
            gradeSelects.forEach(select => {
                if (select.value) {
                    totalGrade += parseInt(select.value);
                    gradeCount++;
                }
            });

            const avgGrade = gradeCount > 0 ? (totalGrade / gradeCount).toFixed(1) : 0;
            let finalGrade = '';
            if (avgGrade >= 0 && avgGrade < 1) finalGrade = '轻度失能';
            else if (avgGrade >= 1 && avgGrade < 2) finalGrade = '中度失能';
            else if (avgGrade >= 2) finalGrade = '重度失能';
            else finalGrade = '未完成评估';

            summaryText += `\n**综合评估结论：**\n`;
            summaryText += `本次评估综合得分为${avgGrade}分，被保险人失能等级评定为：${finalGrade}。\n`;
            // 新增：自评与评估结果对比
            if (selfAssessmentFiles.length > 0) {
                summaryText += `**自评与专业评估对比：** 客户自评等级为E级，专业评估等级为${finalGrade}，${['轻度失能', '中度失能', '重度失能'].includes(finalGrade) ? '符合申请条件' : '暂不符合申请条件'}。`;
            }

            // 更新汇总内容
            summaryContent.textContent = summaryText;
        }

        // 表单验证
        function validateForm() {
            let isValid = true;
            
            // 姓名验证
            const name = document.getElementById('insureeName').value;
            const nameTip = document.getElementById('nameTip');
            if (!name) {
                nameTip.style.display = 'block';
                isValid = false;
            } else {
                nameTip.style.display = 'none';
            }

            // 身份证验证
            const idCard = document.getElementById('idCard').value;
            const idCardTip = document.getElementById('idCardTip');
            const idCardRegex = /^\d{17}[\dXx]$/;
            if (!idCard || !idCardRegex.test(idCard)) {
                idCardTip.style.display = 'block';
                idCardTip.textContent = !idCard ? '请填写身份证号' : '身份证号格式不正确';
                isValid = false;
            } else {
                idCardTip.style.display = 'none';
            }

            // 评估日期验证
            const date = document.getElementById('assessmentDate').value;
            const dateTip = document.getElementById('dateTip');
            if (!date) {
                dateTip.style.display = 'block';
                isValid = false;
            } else {
                dateTip.style.display = 'none';
            }

            // 评估人员验证
            const assessor = document.getElementById('assessor').value;
            const assessorTip = document.getElementById('assessorTip');
            if (!assessor) {
                assessorTip.style.display = 'block';
                isValid = false;
            } else {
                assessorTip.style.display = 'none';
            }

            // 至少完成一项评估
            const gradeSelects = document.querySelectorAll('.grade-select');
            let hasAssessment = false;
            gradeSelects.forEach(select => {
                if (select.value) hasAssessment = true;
            });

            if (!hasAssessment) {
                alert('请至少完成一项评估等级选择！');
                isValid = false;
            }

            return isValid;
        }
